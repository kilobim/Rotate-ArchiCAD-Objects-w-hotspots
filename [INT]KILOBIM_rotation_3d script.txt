!!!!!!!!!!!! PONTOS QUENTES RODAR NO EIXO Y

addy b/2

rotx 90
unID = 1
! --------------------------------------------------------------------------------
! Centro
! --------------------------------------------------------------------------------
hotspot 0, 0, 0, unID, rodar_x, 6     : unID = unID + 1

! --------------------------------------------------------------------------------
! Base
! --------------------------------------------------------------------------------
hotspot A, 0, 0, unID, rodar_x, 4     : unID = unID + 1

! --------------------------------------------------------------------------------
! Ponto móvel
! --------------------------------------------------------------------------------
hotspot A * COS(rodar_x), A * SIN(rodar_x), 0, unID, rodar_x, 5     : unID = unID + 1

! --------------------------------------------------------------------------------
! Referencia
! --------------------------------------------------------------------------------
hotspot 0, 0, 1, unID, rodar_x, 7     : unID = unID + 1

del 2

!!!!!!!!!!!! PONTOS QUENTES RODAR NO EIXO X

addx a/2

rotz 90
rotx -90

! --------------------------------------------------------------------------------
! Centro
! --------------------------------------------------------------------------------
hotspot 0, 0, 0, unID, rodar_y, 6     : unID = unID + 1

! --------------------------------------------------------------------------------
! Base
! --------------------------------------------------------------------------------
hotspot B, 0, 0, unID, rodar_y, 4     : unID = unID + 1

! --------------------------------------------------------------------------------
! Ponto movel
! --------------------------------------------------------------------------------
hotspot B * COS(rodar_y), B * SIN(rodar_y), 0, unID, rodar_y, 5     : unID = unID + 1

! --------------------------------------------------------------------------------
! Referencia
! --------------------------------------------------------------------------------
hotspot 0, 0, 1, unID, rodar_y, 7     : unID = unID + 1

del 3

!!!!!!!!!!!! TRANSFORMAÇÕES

roty -rodar_x

rotx -rodar_y
